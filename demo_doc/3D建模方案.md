**2. Low Poly建筑生成函数**
```javascript
// 创建Low Poly建筑
function createLowPolyBuilding(options) {
  const {
    width = 10,
    depth = 10,
    height = 15,
    color = 0x999999,
    position = { x: 0, y: 0, z: 0 },
    hasRoof = true,
    roofType = 'flat', // flat, pointed, dome
    windows = true,
    windowRows = 3,
    windowCols = 4
  } = options;

  const group = new THREE.Group();

  // 主体
  const geometry = new THREE.BoxGeometry(width, height, depth);
  const material = new THREE.MeshStandardMaterial({
    color: color,
    roughness: 0.8,
    metalness: 0.1,
    flatShading: true // Low Poly关键
  });
  const building = new THREE.Mesh(geometry, material);
  building.position.y = height / 2;
  building.castShadow = true;
  building.receiveShadow = true;
  group.add(building);

  // 屋顶
  if (hasRoof) {
    let roofGeometry, roofMaterial;

    switch (roofType) {
      case 'pointed':
        // 尖顶
        roofGeometry = new THREE.ConeGeometry(
          Math.max(width, depth) * 0.7,
          height * 0.3,
          4
        );
        roofMaterial = new THREE.MeshStandardMaterial({
          color: 0x8B4513,
          flatShading: true
        });
        break;

      case 'dome':
        // 圆顶
        roofGeometry = new THREE.SphereGeometry(
          Math.max(width, depth) * 0.5,
          8, // 减少分段体现Low Poly
          6,
          0, // phiStart
          Math.PI * 0.5 // phiLength
        );
        roofMaterial = new THREE.MeshStandardMaterial({
          color: 0x8B4513,
          flatShading: true
        });
        break;

      default:
        // 平顶
        roofGeometry = new THREE.BoxGeometry(width + 1, 1, depth + 1);
        roofMaterial = new THREE.MeshStandardMaterial({
          color: 0x666666,
          flatShading: true
        });
    }

    const roof = new THREE.Mesh(roofGeometry, roofMaterial);
    roof.position.y = height + (roofType === 'flat' ? 0.5 : 0);
    roof.castShadow = true;
    group.add(roof);
  }

  // 窗户
  if (windows) {
    const windowGeometry = new THREE.PlaneGeometry(1, 1.5);
    const windowMaterial = new THREE.MeshStandardMaterial({
      color: 0x87CEEB,
      emissive: 0x87CEEB,
      emissiveIntensity: 0.2
    });

    for (let row = 0; row < windowRows; row++) {
      for (let col = 0; col < windowCols; col++) {
        const windowMesh = new THREE.Mesh(windowGeometry, windowMaterial);

        // 前面窗户
        windowMesh.position.set(
          (col - (windowCols - 1) / 2) * (width / windowCols),
          (row + 1) * (height / (windowRows + 1)),
          depth / 2 + 0.01
        );
        group.add(windowMesh);

        // 背面窗户
        const windowBack = windowMesh.clone();
        windowBack.position.z = -depth / 2 - 0.01;
        windowBack.rotation.y = Math.PI;
        group.add(windowBack);
      }
    }
  }

  // 设置位置
  group.position.set(position.x, position.y, position.z);

  return group;
}

// 使用示例
const infoBuilding = createLowPolyBuilding({
  width: 20,
  depth: 15,
  height: 25,
  color: 0x4a90d9,
  position: { x: 0, y: 0, z: 0 },
  roofType: 'flat',
  windowRows: 5,
  windowCols: 6
});
scene.add(infoBuilding);
```

```javascript
// 创建地面
function createGround() {
  // 主地面（草地）
  const groundGeometry = new THREE.PlaneGeometry(300, 300);
  const groundMaterial = new THREE.MeshStandardMaterial({
    color: 0x7CBA3D,
    roughness: 0.9,
    flatShading: true
  });
  const ground = new THREE.Mesh(groundGeometry, groundMaterial);
  ground.rotation.x = -Math.PI / 2;
  ground.receiveShadow = true;
  scene.add(ground);

  // 道路
  const roadMaterial = new THREE.MeshStandardMaterial({
    color: 0x555555,
    roughness: 0.9
  });

  // 主干道
  const mainRoad = new THREE.Mesh(
    new THREE.PlaneGeometry(12, 300),
    roadMaterial
  );
  mainRoad.rotation.x = -Math.PI / 2;
  mainRoad.position.y = 0.01;
  mainRoad.receiveShadow = true;
  scene.add(mainRoad);

  // 横向道路
  const crossRoad = new THREE.Mesh(
    new THREE.PlaneGeometry(300, 12),
    roadMaterial
  );
  crossRoad.rotation.x = -Math.PI / 2;
  crossRoad.position.y = 0.01;
  crossRoad.receiveShadow = true;
  scene.add(crossRoad);
}

// 添加植被
function createVegetation() {
  const treeGroup = new THREE.Group();

  // 树干
  const trunkGeometry = new THREE.CylinderGeometry(0.5, 0.8, 4, 6);
  const trunkMaterial = new THREE.MeshStandardMaterial({
    color: 0x8B4513,
    flatShading: true
  });
  const trunk = new THREE.Mesh(trunkGeometry, trunkMaterial);
  trunk.position.y = 2;
  trunk.castShadow = true;
  treeGroup.add(trunk);

  // 树冠（Low Poly风格）
  const foliageGeometry = new THREE.DodecahedronGeometry(3);
  const foliageMaterial = new THREE.MeshStandardMaterial({
    color: 0x228B22,
    flatShading: true
  });
  const foliage = new THREE.Mesh(foliageGeometry, foliageMaterial);
  foliage.position.y = 5;
  foliage.castShadow = true;
  treeGroup.add(foliage);

  return treeGroup;
}

// 在场景中随机添加树木
for (let i = 0; i < 50; i++) {
  const tree = createVegetation();
  tree.position.set(
    (Math.random() - 0.5) * 250,
    0,
    (Math.random() - 0.5) * 250
  );

  // 避开道路
  if (Math.abs(tree.position.x) < 10 || Math.abs(tree.position.z) < 10) {
    continue;
  }

  scene.add(tree);
}
```

## 📸 方案3：摄影测量详细指南

### 第1步：拍摄照片

**设备要求**：
- 手机或相机都可以
- 建议使用分辨率≥12MP

**拍摄技巧**：
```
1. 选择天气晴朗的日子
2. 避免强光直射（早晨或傍晚最佳）
3. 围绕建筑360°拍摄
4. 每隔2-3米拍一张
5. 不同高度拍摄（平视、仰视、俯视）
6. 重叠率≥60%

单个建筑大约需要：
- 小型建筑：30-50张
- 中型建筑：50-100张
- 大型建筑：100-200张
```

**拍摄路线**：
```
       ↑
    ↗  ↑  ↖
   ←  建  →
    ↙  ↓  ↘
（围绕建筑螺旋上升）
```

## ⚡ 性能优化

### 优化策略

**1. 减少面数**
```javascript
// 使用低分段几何体
const geometry = new THREE.BoxGeometry(10, 10, 10, 1, 1, 1); // 最少分段

// 代替
const geometry = new THREE.BoxGeometry(10, 10, 10, 10, 10, 10);
```

**2. 使用实例化**
```javascript
// 对于重复对象（如树木）
const treeGeometry = new THREE.DodecahedronGeometry(3);
const treeMaterial = new THREE.MeshStandardMaterial({ color: 0x228B22 });

const instancedMesh = new THREE.InstancedMesh(
  treeGeometry,
  treeMaterial,
  100 // 100棵树
);

// 设置每棵树的位置
const matrix = new THREE.Matrix4();
for (let i = 0; i < 100; i++) {
  matrix.setPosition(
    (Math.random() - 0.5) * 200,
    5,
    (Math.random() - 0.5) * 200
  );
  instancedMesh.setMatrixAt(i, matrix);
}

scene.add(instancedMesh);
```

**3. 合并几何体**
```javascript
// 对于静态建筑，可以合并
const geometries = [];

buildings.forEach(building => {
  geometries.push(building.geometry);
});

const mergedGeometry = BufferGeometryUtils.mergeGeometries(geometries);
```
